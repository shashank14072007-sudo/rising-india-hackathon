class TemporalDiseaseTransformer(nn.Module):
    """
    Models disease progression across a sequence of images.
    Input: (B, T, C, H, W) — batch of T-frame sequences
    Output: progression_trajectory, prognosis_score, contagion_risk
    """
    def __init__(self, embed_dim=512, n_heads=8, n_layers=4, max_frames=8):
        super().__init__()
        # Per-frame encoder (shared weights)
        self.frame_encoder = TripleBackboneFusion(out_dim=embed_dim)
        # Learnable temporal position embeddings
        self.temporal_pos = nn.Embedding(max_frames, embed_dim)
        # Causal temporal attention (can't attend to future frames)
        encoder_layer = nn.TransformerEncoderLayer(
            d_model=embed_dim, nhead=n_heads,
            dim_feedforward=2048, dropout=0.1, batch_first=True
        )
        self.temporal_transformer = nn.TransformerEncoder(encoder_layer, n_layers)
        # Output heads
        self.trajectory_head = nn.Linear(embed_dim, 64)   # progression vector
        self.prognosis_head  = nn.Linear(embed_dim, 1)    # 48h severity score
        self.contagion_head  = nn.Linear(embed_dim, 1)    # outbreak probability

    def forward(self, frames, timestamps):
        B, T, C, H, W = frames.shape
        # Encode each frame independently
        feats = self.frame_encoder(frames.view(B*T, C, H, W)).view(B, T, -1)
        # Add temporal positional encoding
        pos = self.temporal_pos(timestamps)
        feats = feats + pos
        # Causal mask: each frame attends only to past frames
        causal_mask = nn.Transformer.generate_square_subsequent_mask(T)
        feats = self.temporal_transformer(feats, mask=causal_mask)
        # Use last frame's representation for predictions
        last = feats[:, -1]
        return {
            'trajectory': self.trajectory_head(last),
            'prognosis':  torch.sigmoid(self.prognosis_head(last)),
            'contagion':  torch.sigmoid(self.contagion_head(last))
        }
class GatedTripleFusion(nn.Module):
    """
    Fuses EfficientNet (texture), ViT (global), VMamba (sequential)
    via learned gating — model decides how much to trust each backbone.
    """
    def __init__(self, dim=512):
        super().__init__()
        self.proj_effnet  = nn.Linear(1792, dim)  # EfficientNet-B4 out
        self.proj_vit     = nn.Linear(384,  dim)  # ViT-S/16 out
        self.proj_vmamba  = nn.Linear(512,  dim)  # VMamba out
        # Cross-attention gating: each backbone queries the others
        self.cross_attn = nn.MultiheadAttention(dim, num_heads=8, batch_first=True)
        # Learnable gate weights
        self.gate = nn.Sequential(
            nn.Linear(dim * 3, dim),
            nn.Sigmoid()
        )
        self.out_proj = nn.Linear(dim * 3, dim)

    def forward(self, f_eff, f_vit, f_vmamba, f_sensor=None):
        e = self.proj_effnet(f_eff).unsqueeze(1)
        v = self.proj_vit(f_vit).unsqueeze(1)
        m = self.proj_vmamba(f_vmamba).unsqueeze(1)
        tokens = torch.cat([e, v, m], dim=1)  # (B, 3, dim)
        # Cross-attention: each backbone attends to all others
        attended, _ = self.cross_attn(tokens, tokens, tokens)
        fused = attended.flatten(1)   # (B, 3*dim)
        gate  = self.gate(fused)        # learned importance weights
        out   = self.out_proj(fused) * gate
        if f_sensor is not None:
            out = out + self.sensor_proj(f_sensor)  # late sensor fusion
        return out
class MCDropoutUncertainty:
    """
    Monte Carlo Dropout for epistemic uncertainty estimation.
    Run N stochastic forward passes to get prediction distribution.
    """
    def __init__(self, model, n_passes=30, threshold=0.25):
        self.model = model
        self.n_passes = n_passes
        self.threshold = threshold

    def predict_with_uncertainty(self, x):
        self.model.train()  # keep dropout active
        preds = torch.stack([
            self.model(x)['class_logits']
            for _ in range(self.n_passes)
        ])  # (N, B, C)
        probs = torch.softmax(preds, dim=-1)
        # Epistemic: variance across passes (model uncertainty)
        epistemic = probs.var(dim=0).mean(dim=-1)
        # Aleatoric: mean entropy (data uncertainty)
        mean_probs = probs.mean(dim=0)
        aleatoric  = -(mean_probs * mean_probs.log()).sum(dim=-1)
        self.model.eval()
        return {
            'prediction':    mean_probs.argmax(-1),
            'confidence':    mean_probs.max(-1).values,
            'epistemic_unc': epistemic,
            'aleatoric_unc': aleatoric,
            'needs_review':  epistemic > self.threshold
        }
class ConceptBottleneckModel(nn.Module):
    """
    Human-interpretable concepts learned as intermediate layer.
    Concepts: ['fin_erosion','lesion_density','body_discolor',
               'eye_opacity','gill_inflammation','scale_loss']
    """
    CONCEPTS = ['fin_erosion', 'lesion_density', 'body_discoloration',
                'eye_opacity', 'gill_inflammation', 'scale_loss']

    def __init__(self, backbone_dim=512, n_diseases=20):
        super().__init__()
        self.n_concepts = len(self.CONCEPTS)
        # Step 1: Visual features → concept scores [0,1]
        self.concept_head = nn.Sequential(
            nn.Linear(backbone_dim, 256), nn.ReLU(),
            nn.Linear(256, self.n_concepts), nn.Sigmoid()
        )
        # Step 2: Concept scores → disease prediction
        # ONLY uses concept scores — fully interpretable bottleneck
        self.disease_head = nn.Sequential(
            nn.Linear(self.n_concepts, 64), nn.ReLU(),
            nn.Linear(64, n_diseases)
        )

    def forward(self, features, concept_override=None):
        concept_scores = self.concept_head(features)
        # Optionally let a veterinarian override concept values
        if concept_override is not None:
            concept_scores = concept_override
        disease_logits = self.disease_head(concept_scores)
        return {
            'concepts': dict(zip(self.CONCEPTS, concept_scores[0].tolist())),
            'logits':   disease_logits
        }
